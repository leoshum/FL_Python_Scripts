function GetEffectivePerfSeatsPrices(perfData) {

    var arr = perfData.Prices || [];
    var prices = arr.filter(function (p) {
        var seatIdPrefix = p.FacilityLevel + '-' + p.LogicalLevel;

        var priceIsValid = perfData.Seats.some(function (s) {
            return s.PriceCode !== null &&
                    s.Price !== null &&
                    s.Availability === 2 &&
                    s.SeatId.indexOf(seatIdPrefix) === 0 &&
                    s.PriceCode.trim() === p.PriceCode.trim() &&
                    s.PriceSet === p.PriceSet
        });

        return priceIsValid;
    });

    return prices;
}
